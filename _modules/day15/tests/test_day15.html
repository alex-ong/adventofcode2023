def test_parse_pt2() -> None:
    """Test parsing input for part2."""
    steps: list[str] = get_input(INPUT_SMALL)
    step: Step = parse_step_pt2(steps[0])
    assert step.lens_name == "rn"
    assert step.process == AddRemove.Add
